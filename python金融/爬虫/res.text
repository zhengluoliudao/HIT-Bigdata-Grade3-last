$(window).scroll((function(){
    var head=$("#head"),
        w=$(window);
    var downPos=20;
    var timeoutid;
    var oldHeadTop=head.offset().top;
    var onscroll=function(){
        if(timeoutid){
            clearTimeout(timeoutid);
            timeoutid=false;
        }
        timeoutid=setTimeout(function(){
            var scrollTop=w.scrollTop();
            if(scrollTop>downPos+oldHeadTop){
                timeoutid=setTimeout(function(){
                    head.addClass("s_down");
                },0);
            }else if(scrollTop<=downPos+oldHeadTop){
                timeoutid=setTimeout(function(){
                    head.removeClass("s_down");
                },0);
            }
        },50);
    };
    onscroll();
    return onscroll;
})());
